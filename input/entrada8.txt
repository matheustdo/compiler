struct S {
	const {
		int cadastro = 16;
		real cr[10] = {10,10};

		cadastro = 1;
		cr[7] = 2.0;
	}
}

struct T extends struct S {
	var {
		int vi;
		real vr[10];
	}
}

var {
	int vi;
	real vr[10];

	typedef struct S S;
	typedef struct T T;

	S s[10][10];
	T t[10][10];
}

function real sum(real vr[]) {
	var {
		int i;
		real s;

		i = 0;
		s = 0.0;
	}

	while(i < 10)
		s = s + vr[i];
	
	return s;
}
